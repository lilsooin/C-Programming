### extern keyword
- use `extern` 
```c
/* repo.h */
extern int g_count;
void add_something(void);
```


```c
/* repo.c */
#include "repo.h"

int g_count = 0;
void add_something(void)
{
    ++g_count;
}
```


```c
/* main.c */
#include <stdio.h>
#include "repo.h"

/* or use it in main.c */
/* extern int g_count; */

int main(void)
{
    add_something();
    printf("# added something: %d\n", g_count);
}

return 0;
```
